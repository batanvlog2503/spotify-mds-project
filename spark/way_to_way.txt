Phần 3 — Đọc stream từ Kafka:
pythonraw_df = spark.readStream \
    .format("kafka") \
    .option("kafka.bootstrap.servers", "kafka:9092") \
    .option("subscribe", "spotify-events") \
    .option("startingOffsets", "earliest") \
    .option("failOnDataLoss", "false") \
    .load()

# Kafka trả về binary → parse thành JSON theo schema
parsed_df = raw_df.select(
    from_json(
        col("value").cast("string"),
        schema
    ).alias("data"),
    col("timestamp").alias("kafka_timestamp")
).select("data.*", "kafka_timestamp") \
 .withColumn("processed_at", current_timestamp())
Phần 4 — Tách clean / dirty:
python# Điều kiện dữ liệu sạch
clean_condition = (
    col("event_id").isNotNull() &
    col("user_id").isNotNull()  &
    col("song_id").isNotNull()  &
    col("timestamp").isNotNull()
)

clean_df = parsed_df.filter(clean_condition)

# Dữ liệu bẩn + gắn lý do
dirty_df = parsed_df.filter(~clean_condition) \
    .withColumn("_reject_reason",
        when(col("event_id").isNull(),  lit("missing_event_id"))
       .when(col("user_id").isNull(),   lit("missing_user_id"))
       .when(col("song_id").isNull(),   lit("missing_song_id"))
       .when(col("timestamp").isNull(), lit("missing_timestamp"))
       .otherwise(lit("unknown"))
    ) \
    .withColumn("_rejected_at", current_timestamp().cast("string"))
Phần 5 — Ghi xuống MinIO:
python# Ghi clean stream
clean_query = clean_df.writeStream \
    .format("json") \
    .option("path",              "s3a://your-bucket/bronze/clean/") \
    .option("checkpointLocation","s3a://your-bucket/checkpoint/clean/") \
    .partitionBy("country") \
    # partition theo country → query nhanh hơn trong dbt
    .outputMode("append") \
    .trigger(processingTime="30 seconds") \
    # cứ 30 giây xử lý 1 micro-batch
    .start()

# Ghi dirty stream
dirty_query = dirty_df.writeStream \
    .format("json") \
    .option("path",              "s3a://your-bucket/bronze/quarantine/") \
    .option("checkpointLocation","s3a://your-bucket/checkpoint/quarantine/") \
    .outputMode("append") \
    .trigger(processingTime="30 seconds") \
    .start()

# Giữ 2 stream chạy mãi
spark.streams.awaitAnyTermination()

    - minio

    - ./spark:/opt/spark-jobs
  command: >
    spark-submit
    --master spark://spark-master:7077
    --packages org.apache.spark:spark-sql-kafka-0-10_2.12:3.5.0,org.apache.hadoop:hadoop-aws:3.3.4
